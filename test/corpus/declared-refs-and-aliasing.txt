==================
declared_refs and refaliasing
==================
# @id: refs.declared.001
# @tags: feature declared_refs refaliasing experimental
# @perl: 5.26+
# @flags: experimental
use feature 'declared_refs';
use feature 'refaliasing';
my \($x, $y) = \($a, $b);
\my $scalar = \$other;
\my @array = \@other;

==================
refaliasing with no feature
==================
# @id: refs.aliasing.001
# @tags: refaliasing experimental
# @perl: 5.22+
# @flags: experimental
use experimental 'refaliasing';
\$x = \$y;
\@a = \@b;
\%h = \%g;
foreach \my $item (@list) {
    $item++;
}

==================
declared refs in signatures
==================
# @id: refs.signatures.001
# @tags: declared_refs signatures experimental
# @perl: 5.26+
# @flags: experimental
use feature 'declared_refs', 'signatures';
sub take_ref (\$x) {
    $$x++;
}
my $val = 42;
take_ref($val);
