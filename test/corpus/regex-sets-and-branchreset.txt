==================
Character class set operations
==================
# @id: regex.sets.001
# @tags: regex sets character-class experimental
# @perl: 5.18+
# @flags: experimental
use experimental 'regex_sets';
"a" =~ /(?[[a-z] & \w])/;
"5" =~ /(?[[0-9] - [13579]])/;
"x" =~ /(?[[a-z] + [A-Z]])/;
"!" =~ /(?[[[:punct:]] & [!-/]])/;

==================
Branch reset groups
==================
# @id: regex.branchreset.001
# @tags: regex branch-reset capture
# @perl: 5.10+
"ab" =~ /^(?|(a)|(ab))(?|b|)$/;
"test" =~ /(?|(t)(e)|(te))(st)/;
my $match = "foo" =~ /(?|(f)(o+)|(fo)(o))/;

==================
Complex set operations
==================
# @id: regex.sets.002
# @tags: regex sets union intersection subtraction
# @perl: 5.18+
# @flags: experimental
use experimental 'regex_sets';
/(?[ [a-z] & [^aeiou] ])/;
/(?[ [\d\w] - [\d] ])/;
/(?[ [[:alpha:]] + [[:digit:]] ])/;
/(?[ ( [a-m] + [n-z] ) & [aeiou] ])/;

==================
Named capture with branch reset
==================
# @id: regex.branchreset.002
# @tags: regex branch-reset named-capture
# @perl: 5.10+
"test" =~ /(?|(?<word>t..)(?<end>t)|(?<word>te)(?<end>st))/;
my $word = $+{word};
my $end = $+{end};
