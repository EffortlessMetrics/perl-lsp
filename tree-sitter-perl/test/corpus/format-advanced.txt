==================
Format write and formline
==================

local $~ = "MYFORMAT";
write;

write FILEHANDLE;

formline <<'END', @values;
@<<<< @>>>> @||||
END

my $picture = '@<<<< @>>>>';
formline $picture, $left, $right;
my $result = $^A;
$^A = "";

swrite(<<'END', $name, $value);
Name: @<<<<<<<<< Value: @>>>>
END

sub swrite {
    my $format = shift;
    $^A = "";
    formline($format, @_);
    return $^A;
}
